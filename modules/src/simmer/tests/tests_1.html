[docs]	def test_create_im_default(self):
		print('Testing default image creation')
		raw_dir, reddir = os.getcwd() + '/src/simmer/tests/sky_test/', os.getcwd() + '/src/simmer/tests/sky_test/'
		s_dir = raw_dir
		compare_final_im = pyfits.getdata(raw_dir + 'Ks/compare_final_im_default.fits', 0)
